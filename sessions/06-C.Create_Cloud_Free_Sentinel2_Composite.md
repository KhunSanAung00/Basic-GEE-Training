## 2. Setup Parameters

Define the necessary parameters for the image collection and cloud filtering.

### 2.1 Config
```javascript
// Define the start and end date for the image collection
var START_DATE = '2024-05-01';
var END_DATE = '2024-10-31';

// Define the maximum percentage of cloud cover allowed in images
var CLOUD_FILTER = 70;

// Thresholds for cloud probability and dark pixels
var CLD_PRB_THRESH = 40; // Cloud probability threshold
var NIR_DRK_THRESH = 0.15; // Near-infrared dark pixel threshold

// Distance to project cloud shadows (in pixels)
var CLD_PRJ_DIST = 2;

// Buffer distance to dilate cloud/shadow mask (in pixels)
var BUFFER = 50;
```

## 3. Helper Functions to Get Sentinel-2 Cloud-Free Image Collection

### 3.1 Join Sentinel-2 SR with Sentinel-2 cloud probability

```javascript
function get_s2_sr_cld_col(aoi, startDate, endDate) {
  var s2_sr_col = ee.ImageCollection("COPERNICUS/S2_SR_HARMONIZED")
    .filterBounds(aoi)
    .filterDate(startDate, endDate)
    .filter(ee.Filter.lte('CLOUDY_PIXEL_PERCENTAGE', CLOUD_FILTER));

  var s2_cloudless_col = ee.ImageCollection('COPERNICUS/S2_CLOUD_PROBABILITY')
    .filterBounds(aoi)
    .filterDate(startDate, endDate);

  return ee.ImageCollection(ee.Join.saveFirst('s2cloudless').apply({
    primary: s2_sr_col,
    secondary: s2_cloudless_col,
    condition: ee.Filter.equals({
      leftField: 'system:index',
      rightField: 'system:index'
    })
  }));
}
```


### 3.2 Shadow bands

```javascript
// Add cloud shadow bands
function add_shadow_bands(image) {
  var notWater = image.select('SCL').neq(6);

  var darkPixels = image.select('B8')
    .lt(NIR_DRK_THRESH * 1e4)
    .and(notWater)
    .rename('dark_pixels');

  var shadowAzimuth = ee.Number(90)
    .subtract(ee.Number(image.get('MEAN_SOLAR_AZIMUTH_ANGLE')));

  var cloudProjection = image.select('clouds')
    .directionalDistanceTransform(shadowAzimuth, CLD_PRJ_DIST * 10)
    .reproject({
      crs: image.select(0).projection(),
      scale: 100
    })
    .select('distance')
    .mask()
    .rename('cloud_transform');

  var shadows = cloudProjection
    .and(darkPixels)
    .rename('shadows');

  return image.addBands(darkPixels)
              .addBands(cloudProjection)
              .addBands(shadows);
}
```


### 3.3 Create Mask from Shadow & Cloud Mask 

```javascript
// Create final cloud and shadow mask
function add_cld_shdw_mask(image) {
  image = add_cloud_bands(image);
  image = add_shadow_bands(image);

  var cloudShadowMask = image.select('clouds')
    .add(image.select('shadows'))
    .gt(0);

  cloudShadowMask = cloudShadowMask
    .focalMin(2)
    .focalMax(BUFFER / 10)
    .reproject({
      crs: image.select(0).projection(),
      scale: 20
    })
    .rename('cloud_shadow_mask');

  return image.addBands(cloudShadowMask);
}
```


### 3.4 Apply Final Mask band to the image

```javascript
// Apply cloud and shadow mask
function apply_cld_shdw_mask(image) {
  var mask = image.select('cloud_shadow_mask').not();

  return image
    .select(['B2', 'B3', 'B4', 'B8', 'B11', 'B12'])
    .updateMask(mask);
}

```



## 4. Get Sentinel-2 collection

```javascript
var s2 = get_s2_sr_cld_col(AOI, START_DATE, END_DATE);

print('Original joined collection:', s2);
print('Number of images:', s2.size());
```

## 5. Apply cloud/shadow masking
```javascript
var s2_masked = s2
  .map(add_cld_shdw_mask)
  .map(apply_cld_shdw_mask);

print('Cloud masked collection:', s2_masked);
```
